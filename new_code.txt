#include <iostream>
#include <string>
#include <stack>
#include <sstream>
using namespace std;
int evaluateExpression(string expression) {
    stack<char> operators;
    stack<int> operands;
    stringstream ss(expression);
    string current;
    while (ss >> current) {  // Use '>>' to read the entire number
        if (isdigit(current[0])) {
            operands.push(stoi(current));
        } else if (current == "(") {
            operators.push(current[0]);
        } else if (current == "+" || current == "-" || current == "*" || current == "/") {
            while (!operators.empty() && operators.top() != '(' && ((current[0] == '*' || current[0] == '/') || (operators.top() == '+' || operators.top() == '-'))) {
                int b = operands.top();
                operands.pop();
                int a = operands.top();
                operands.pop();
                char op = operators.top();
                operators.pop();
                int result = 0;
                if (op == '+') {
                    result = a + b;
                } else if (op == '-') {
                    result = a - b;
                } else if (op == '*') {
                    result = a * b;
                } else if (op == '/') {
                    result = a / b;
                }
                operands.push(result);
            }
            operators.push(current[0]);
        } else if (current == ")") {
            while (!operators.empty() && operators.top() != '(') {
                int b = operands.top();
                operands.pop();
                int a = operands.top();
                operands.pop();
                char op = operators.top();
                operators.pop();
                int result = 0;
                if (op == '+') {
                    result = a + b;
                } else if (op == '-') {
                    result = a - b;
                } else if (op == '*') {
                    result = a * b;
                } else if (op == '/') {
                    result = a / b;
                }
                operands.push(result);
            }
            operators.pop(); // Remove the '('
        }
    }
    while (!operators.empty()) {
        int b = operands.top();
        operands.pop();
        int a = operands.top();
        operands.pop();
        char op = operators.top();
        operators.pop();
        int result = 0;
        if (op == '+') {
            result = a + b;
        } else if (op == '-') {
            result = a - b;
        } else if (op == '*') {
            result = a * b;
        } else if (op == '/') {
            result = a / b;
        }
        operands.push(result);
    }
    return operands.top();
}
int main() {
    cout << "Введите выражение:" << endl;
    string input;
    getline(cin, input);
    int result = evaluateExpression(input);
    cout << "Результат: " << result << endl;
    return 0;
}
